SQL CẦN CHẠY THÊM (setup-v4.sql) 
-- Lưu Groq key vào profile
ALTER TABLE profiles ADD COLUMN IF NOT EXISTS groq_api_key TEXT DEFAULT '';

-- Bảng ôn câu sai
CREATE TABLE IF NOT EXISTS wrong_answers (
  user_id UUID REFERENCES profiles(id) ON DELETE CASCADE,
  question_id UUID REFERENCES questions(id) ON DELETE CASCADE,
  wrong_count INTEGER DEFAULT 1,
  last_wrong_at TIMESTAMPTZ DEFAULT NOW(),
  PRIMARY KEY (user_id, question_id)
);
ALTER TABLE wrong_answers ENABLE ROW LEVEL SECURITY;
CREATE POLICY "wa_own" ON wrong_answers USING (auth.uid() = user_id);

-- Rubric scores cho vấn đáp
ALTER TABLE oral_results ADD COLUMN IF NOT EXISTS rubric_scores JSONB DEFAULT '{}';
